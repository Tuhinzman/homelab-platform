#   sed strip resourceVersion/uid/creationTimestamp from backup

# === Block added to .data.Corefile (after the existing .:53 block) ===

lab:53 {
    errors
    cache 30
    forward . 10.10.0.50
    reload
}

# === Plugin breakdown ===
#
# errors      — log errors to stderr (visible via `kubectl logs`)
# cache 30    — 30s positive/negative cache (matches default block, reduces svc-1 load)
# forward . 10.10.0.50    — forward all queries within `lab` zone to svc-1's dnsmasq
# reload      — auto-detect Corefile changes and reload without restart
#
# Plugins NOT in this block (and why):
#   health, ready, prometheus    — root-block concerns; this block is pure forwarder
#   loop                         — root block already has it; loop check at root level only
#   kubernetes                   — irrelevant for non-cluster.local queries
#   loadbalance                  — single upstream, no balancing needed
